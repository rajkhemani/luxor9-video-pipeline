   if (!threshold) return 0;        const { scrollStart, scrollEnd } = threshold;    const localProgress = (globalProgress - scrollStart) / (scrollEnd - scrollStart);        return Math.max(0, Math.min(1, localProgress));  }    /**   * Easing functions for different transition types   */  export const easings = {    intelligence: (t: number): number => {      // Custom easing: fast start, slow middle, sharp end      return t < 0.5        ? 4 * t * t * t        : 1 - Math.pow(-2 * t + 2, 3) / 2;    },        neural: (t: number): number => {      // Smooth deceleration      return 1 - Math.pow(1 - t, 4);    },        decision: (t: number): number => {      // Sharp acceleration at the end (decision moment)      return t * t * t;    },  };    /**   * Interpolates between values based on scroll progress   */  export function interpolate(    progress: number,    from: number,    to: number,    easing: keyof typeof easings = 'neural'  ): number {    const easedProgress = easings[easing](progress);    return from + (to - from) * easedProgress;  }    /**   * Interpolates 3D positions   */  export function interpolatePosition(    progress: number,    from: [number, number, number],    to: [number, number, number],    easing: keyof typeof easings = 'neural'  ): [number, number, number] {    const easedProgress = easings[easing](progress);  import { useMemo, useRef } from 'react';  import * as THREE from 'three';    interface AgentSwarmProps {    count: number;    radius: number;    corePosition: [number, number, number];  }    export function AgentSwarm({ count, radius, corePosition }: AgentSwarmProps) {    const meshRef = useRef<THREE.InstancedMesh>(null);        const agents = useMemo(() => {      return Array.from({ length: count }, (_, i) => ({        id: i,        orbitSpeed: 0.2 + Math.random() * 0.3,        orbitRadius: radius * (0.8 + Math.random() * 0.4),        orbitPhase: Math.random() * Math.PI * 2,        orbitTilt: (Math.random() - 0.5) * Math.PI * 0.5,      }));    }, [count, radius]);        useFrame(({ clock }) => {      if (!meshRef.current) return;            const time = clock.getElapsedTime();      const matrix = new THREE.Matrix4();      const position = new THREE.Vector3();            agents.forEach((agent, i) => {        const angle = time * agent.orbitSpeed + agent.orbitPhase;                position.set(          Math.cos(angle) * agent.orbitRadius,          Math.sin(agent.orbitTilt) * Math.sin(angle) * agent.orbitRadius * 0.5,          Math.sin(angle) * agent.orbitRadius        );                position.add(new THREE.Vector3(...corePosition));        matrix.setPosition(position);        meshRef.current!.setMatrixAt(i, matrix);      });            meshRef.current.instanceMatrix.needsUpdate = true;    });        return (      <instancedMesh ref={meshRef} args={[undefined, undefined, count]}>        <sphereGeometry args={[0.05, 16, 16]} />        <meshStandardMaterial          color="#6E56FF"          emissive="#6E56FF"          emissiveIntensity={0.5}        />      </instancedMesh>    );  }  ```    ### 26.3 Particle Flow Shader    ```glsl  // particle.vert  uniform float uTime;  uniform float uProgress;    attribute float aSpeed;  attribute float aOffset;  attribute vec3 aEndPosition;    varying float vAlpha;  varying float vSpeed;    void main() {    float progress = mod(uTime * aSpeed + aOffset, 1.0);        vec3 pos = mix(position, aEndPosition, progress);        // Add turbulence    pos.x += sin(uTime * 2.0 + aOffset * 10.0) * 0.1;    pos.y += cos(uTime * 1.5 + aOffset * 8.0) * 0.1;        vAlpha = sin(progress * 3.14159);    vSpeed = aSpeed;        vec4 mvPosition = modelViewMatrix * vec4(pos, 1.0);    gl_PointSize = (1.0 / -mvPosition.z) * 50.0;    gl_Position = projectionMatrix * mvPosition;  }    // particle.frag  varying float vAlpha;  varying float vSpeed;    void main() {    float dist = length(gl_PointCoord - vec2(0.5));    if (dist > 0.5) discard;        float alpha = smoothstep(0.5, 0.0, dist) * vAlpha;    vec3 color = mix(vec3(0.0, 0.83, 0.67), vec3(0.43, 0.34, 1.0), vSpeed);        gl_FragColor = vec4(color, alpha);  }  ```    ---    ## 27.
